351 
352  # check for any consistent hypothesis
353  hplist = ( client_find_consistent_hyp( ) ).hyp
354  if ( len( hplist ) == 0 ):
355  return 'no_consistent_hp'
356 
357  rospy.loginfo( "[state:reasoning] complete hypotheses found: %d", len( hplist ) )

360 
361  # choose random one among the complete hypotheses

363 
364  rospy.loginfo( "[state:reasoning] selected hyp. with tag: %s | actual position: %s", to_evaluate.tag, actual_position )
365 
366  # store the hypothesis

369 
370  # choose the outcome
371  if ( actual_position == charge.where ):
372  return 'ready_for_charge'
