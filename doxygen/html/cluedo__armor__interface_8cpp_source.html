49 // perform difference between the first array and the intersection between the two arrays
50 std::vector<std::string> PerformDifferenceBetween( std::vector<std::string> list1, std::vector<std::string> list2 )
51 {
52  std::vector<std::string> to_return;
53 
54  for( auto it1 = list1.begin(); it1 != list1.end(); ++it1 )
55  {
56  bool found = false;
57  for( auto it2 = list2.begin(); it2 != list2.end(); ++it2 )
58  {
59  if( *it2 == *it1 )
60  {
61  found = true;
62  break;
63  }
64  }
65 
66  if( !found )
67  to_return.push_back( *it1 );
68  }
69 
70  return to_return;
71 }
