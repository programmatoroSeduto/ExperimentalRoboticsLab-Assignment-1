307 void generateMystery( std::vector<std::string> list_who, std::vector<std::string> list_where, std::vector<std::string> list_what, int tot_hints )
308 {
309  ROS_INFO_STREAM( OUTLABEL << "case generation started " );
310 
311  // shuffle the arrays before starting
312  std::random_shuffle( list_who.begin(), list_who.end() );
313  std::random_shuffle( list_where.begin(), list_where.end() );
314  std::random_shuffle( list_what.begin(), list_what.end() );
315 
316  // generate the solution without the ID
317  solution_where.HintType = "where";
318  solution_where.HintContent = chooseHintFrom( list_where );
319 
320  solution_who.HintType = "who";
321  solution_who.HintContent = chooseHintFrom( list_who );
322 
323  solution_what.HintType = "what";
324  solution_what.HintContent = chooseHintFrom( list_what );
325 
326  ROS_INFO_STREAM( OUTLABEL << "the solution is " << "(where:" << solution_where.HintContent << ", who:" << solution_who.HintContent << ", what:" << solution_what.HintContent << ")" );
327 
328  // generate the ID of the solution
329  int solutionID = randomIndex( MAX_NUM_HINTS-1 );
330  solution_who.HintID = solutionID;
331  solution_where.HintID = solutionID;
332  solution_what.HintID = solutionID;
333 
334  ROS_INFO_STREAM( OUTLABEL << "the solution has ID:" << solutionID );
335 
336  // generate the case
337  for( int i=0; i<tot_hints; ++i )
338  {
339  if( i == solutionID )
340  {
341  mysterylist.push_back( solution_what );
342  mysterylist.push_back( solution_where );
343  mysterylist.push_back( solution_who );
344 
345  continue;
346  }
347  else
348  {
349  for( int j=0; j<MAX_SIZE_HINT; ++j )
350  {
351  robocluedo_msgs::Hint h;
352  h.HintID = i;
353  switch( randomIndex( 5 ) )
354  {
355  case 0:
356  h.HintType = "who";
357  h.HintContent = chooseHintFrom( list_who );
358  break;
359  case 1:
360  h.HintType = "where";
361  h.HintContent = chooseHintFrom( list_where );
362  break;
363  case 2:
364  h.HintType = "what";
365  h.HintContent = chooseHintFrom( list_what );
366  break;
367  default:
368  continue;
369  break;
370  }
371 
372  mysterylist.push_back( h );
373  }
374  }
375  }
376 
377  ROS_INFO_STREAM( OUTLABEL << "hints generation finished. Generated: " << mysterylist.size() << " hints" );
378 
379  // final shuffle
380  std::random_shuffle( mysterylist.begin(), mysterylist.end() );
381 }
