279 void generateMystery( std::vector<std::string> list_who, std::vector<std::string> list_where, std::vector<std::string> list_what, int tot_hints )
280 {
281  ROS_INFO_STREAM( OUTLABEL << "case generation started " );
282 
283  // shuffle the arrays before starting
284  std::random_shuffle( list_who.begin(), list_who.end() );
285  std::random_shuffle( list_where.begin(), list_where.end() );
286  std::random_shuffle( list_what.begin(), list_what.end() );
287 
288  // generate the solution without the ID
289  solution_where.HintType = "where";
290  solution_where.HintContent = chooseHintFrom( list_where );
291 
292  solution_who.HintType = "who";
293  solution_who.HintContent = chooseHintFrom( list_who );
294 
295  solution_what.HintType = "what";
296  solution_what.HintContent = chooseHintFrom( list_what );
297 
298  ROS_INFO_STREAM( OUTLABEL << "the solution is " << "(where:" << solution_where.HintContent << ", who:" << solution_who.HintContent << ", what:" << solution_what.HintContent << ")" );
299 
300  // generate the ID of the solution
301  int solutionID = randomIndex( MAX_NUM_HINTS-1 );
302  solution_who.HintID = solutionID;
303  solution_where.HintID = solutionID;
304  solution_what.HintID = solutionID;
305 
306  ROS_INFO_STREAM( OUTLABEL << "the solution has ID:" << solutionID );
307 
308  // generate the case
309  for( int i=0; i<tot_hints; ++i )
310  {
311  if( i == solutionID )
312  {
313  mysterylist.push_back( solution_what );
314  mysterylist.push_back( solution_where );
315  mysterylist.push_back( solution_who );
316 
317  continue;
318  }
319  else
320  {
321  for( int j=0; j<MAX_SIZE_HINT; ++j )
322  {
323  robocluedo_msgs::Hint h;
324  h.HintID = i;
325  switch( randomIndex( 5 ) )
326  {
327  case 0:
328  h.HintType = "who";
329  h.HintContent = chooseHintFrom( list_who );
330  break;
331  case 1:
332  h.HintType = "where";
333  h.HintContent = chooseHintFrom( list_where );
334  break;
335  case 2:
336  h.HintType = "what";
337  h.HintContent = chooseHintFrom( list_what );
338  break;
339  default:
340  continue;
341  break;
342  }
343 
344  mysterylist.push_back( h );
345  }
346  }
347  }
348 
349  ROS_INFO_STREAM( OUTLABEL << "hints generation finished. Generated: " << mysterylist.size() << " hints" );
350 
351  // final shuffle
352  std::random_shuffle( mysterylist.begin(), mysterylist.end() );
353 }
