118 std::vector<std::string> PerformDifferenceBetween( std::vector<std::string> list1, std::vector<std::string> list2 )
119 {
120  std::vector<std::string> to_return;
121 
122  for( auto it1 = list1.begin(); it1 != list1.end(); ++it1 )
123  {
124  bool found = false;
125  for( auto it2 = list2.begin(); it2 != list2.end(); ++it2 )
126  {
127  if( *it2 == *it1 )
128  {
129  found = true;
130  break;
131  }
132  }
133 
134  if( !found )
135  to_return.push_back( *it1 );
136  }
137 
138  return to_return;
139 }

155 bool ServiceAddHint( robocluedo_msgs::AddHint::Request& hint, robocluedo_msgs::AddHint::Response& success )

162  if( !armor->ExistsIndiv( hypname ) )
163  armor->AddIndiv( hypname, "HYPOTHESIS", false );
164 
165  // if the Belem is not defined, add it

170  else if( hint.property == "who" )
171  armor->AddIndiv( hint.Belem, "PERSON" );
172  else if( hint.property == "what" )
173  armor->AddIndiv( hint.Belem, "WEAPON" );
174  else
175  {
176  OUTLOG( "ERROR: not a valid hint property." );
177  success.success = false;

179  return true;

181  }
182 
183  // add the predicate
184  OUTLOG( "set object property " << "(" << hypname << ", " << hint.Belem << "):" << hint.property );
185  armor->SetObjectProperty( hint.property, hypname, hint.Belem );
186 
187  // perform the update
188  armor->UpdateOntology( );
189 
190  success.success = true;
