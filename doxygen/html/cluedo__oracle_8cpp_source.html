155 // generate the solution of the case
156 void generateMystery( std::vector<std::string> list_who, std::vector<std::string> list_where, std::vector<std::string> list_what, int tot_hints )
157 {
158  ROS_INFO_STREAM( OUTLABEL << "case generation started " );
159 
160  // shuffle the arrays before starting
161  std::random_shuffle( list_who.begin(), list_who.end() );
162  std::random_shuffle( list_where.begin(), list_where.end() );
163  std::random_shuffle( list_what.begin(), list_what.end() );
164 
165  // generate the solution without the ID
166  solution_where.HintType = "where";
167  solution_where.HintContent = chooseHintFrom( list_where );
168 
169  solution_who.HintType = "who";
170  solution_who.HintContent = chooseHintFrom( list_who );
171 
172  solution_what.HintType = "what";
173  solution_what.HintContent = chooseHintFrom( list_what );
174 
175  ROS_INFO_STREAM( OUTLABEL << "the solution is " << "(where:" << solution_where.HintContent << ", who:" << solution_who.HintContent << ", what:" << solution_what.HintContent << ")" );
176 
177  // generate the ID of the solution
178  int solutionID = randomIndex( MAX_NUM_HINTS-1 );
179  solution_who.HintID = solutionID;
180  solution_where.HintID = solutionID;
181  solution_what.HintID = solutionID;
182 
183  ROS_INFO_STREAM( OUTLABEL << "the solution has ID:" << solutionID );
184 
185  /*
186  * for MAX_NUM_HINTS times:
187  * generate a number from 3 to MAX_SIZE_HINT
188  * for the number choosen before:
189  * choose one of the lists (number from 0 to 2)
190  * take the element from the selected list
191  * type and value
192  * and put it into the mystery list in the right cell
193  *
194  * if the ID belongs to the solution, insert it instead of another new random hint
195  */
196  for( int i=0; i<tot_hints; ++i )
197  {
198  if( i == solutionID )
199  {
200  mysterylist.push_back( solution_what );
201  mysterylist.push_back( solution_where );
202  mysterylist.push_back( solution_who );
203 
204  continue;
205  }
206  else
207  {
208  for( int j=0; j<MAX_SIZE_HINT; ++j )
209  {
210  robocluedo_msgs::Hint h;
211  h.HintID = i;
212  switch( randomIndex( 5 ) )
213  {
214  case 0:
215  h.HintType = "who";
216  h.HintContent = chooseHintFrom( list_who );
217  break;
218  case 1:
219  h.HintType = "where";
220  h.HintContent = chooseHintFrom( list_where );
221  break;
222  case 2:
223  h.HintType = "what";
224  h.HintContent = chooseHintFrom( list_what );
225  break;
226  default:
227  continue;
228  break;
229  }
230 
231  mysterylist.push_back( h );
232  }
233  }
234  }
235 
236  ROS_INFO_STREAM( OUTLABEL << "hints generation finished. Generated: " << mysterylist.size() << " hints" );
237 
238  // final shuffle
239  std::random_shuffle( mysterylist.begin(), mysterylist.end() );
240 }
